function esValidoSudoku(sudoku) {
    // Verificar filas
    for (let i = 0; i < 9; i++) {
        let filaSet = new Set();
        for (let j = 0; j < 9; j++) {
            if (sudoku[i][j] === 0 || filaSet.has(sudoku[i][j])) {
                return false;
            }
            filaSet.add(sudoku[i][j]);
        }
    }
    
    // Verificar columnas
    for (let i = 0; i < 9; i++) {
        let columnaSet = new Set();
        for (let j = 0; j < 9; j++) {
            if (sudoku[j][i] === 0 || columnaSet.has(sudoku[j][i])) {
                return false;
            }
            columnaSet.add(sudoku[j][i]);
        }
    }
    
    // Verificar regiones 3x3
    for (let i = 0; i < 9; i += 3) {
        for (let j = 0; j < 9; j += 3) {
            let regionSet = new Set();
            for (let k = i; k < i + 3; k++) {
                for (let l = j; l < j + 3; l++) {
                    if (sudoku[k][l] === 0 || regionSet.has(sudoku[k][l])) {
                        return false;
                    }
                    regionSet.add(sudoku[k][l]);
                }
            }
        }
    }
    
    return true;
}
