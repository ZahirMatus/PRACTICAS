// Función para contar cuántos números son mayores que x en la lista
function contarMayoresQueX(lista, x) {
    let conteo = 0;
    for (let i = 0; i < lista.length; i++) {
        if (lista[i] > x) {
            conteo++;
        }
    }
    return conteo;
}

// Obtener la cantidad de números
const n = parseInt(prompt("Ingresa cuántos números quieres leer: "));

// Leer los números
const numeros = [];
console.log("Ingresa los números:");
for (let i = 0; i < n; i++) {
    const numero = parseFloat(prompt());
    numeros.push(numero);
}

// Obtener el valor x con el que comparar
const x = parseFloat(prompt("Ingresa el valor x con el cual comparar: "));

// Contar números mayores que x
const cantidadMayores = contarMayoresQueX(numeros, x);
console.log(`De tu lista, ${cantidadMayores} valores son mayores a ${x}`);
