// Función para obtener la suma de los elementos en un array
function sumArray(arr) {
  return arr.reduce((acc, curr) => acc + curr, 0);
}

// Función para obtener la cantidad de datos mayores que el promedio
function datosMayoresAlPromedio(datos) {
  const suma = sumArray(datos);
  const promedio = suma / datos.length;
  const mayoresAlPromedio = datos.filter((dato) => dato > promedio);
  return mayoresAlPromedio.length;
}
